%% unevaluated leaflet code, create map
\begin{Schunk}
\begin{Sinput}
> m <- leaflet() %>%
+   addTiles() %>%
+   addMarkers(lng = -77.03673, lat = 38.89761)
> m
\end{Sinput}
\end{Schunk}
